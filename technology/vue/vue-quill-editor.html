# vue-quill-editor

前言：在项目中遇到一个需要富文本编辑器的功能，上午查看发现vue-quill-editor这个插件好评挺多的，github点赞量也挺高的，所以我也就直接使用这个编辑器了

  1. 先安装vue-quill-editor和quill
npm install vue-quill-editor quill -S

2.全局引入

import { quillEditor } from 'vue-quill-editor' // 调用富文本编辑器
import 'quill/dist/quill.snow.css' // 富文本编辑器外部引用样式  三种样式三选一引入即可
//import 'quill/dist/quill.core.css'
//import 'quill/dist/quill.bubble.css'
import * as Quill from 'quill'; // 富文本基于quill

3.使用

<template>
<quill-editor
  v-model="content" // 编辑框中的内容
  ref="myQuillEditor"
  :options="editorOption" // 编辑器toolbar配置项
  @focus="onEditorFocus($event)" // 聚焦事件
  @blur="onEditorBlur($event)" // 失焦事件
  @change="onEditorChange($event)"> // 改变事件
</quill-editor>
</template>

<script>
import { quillEditor } from "vue-quill-editor"; //调用编辑器
import 'quill/dist/quill.core.css';
import 'quill/dist/quill.snow.css';
import 'quill/dist/quill.bubble.css';
export default {
    components: {
        quillEditor
    },
    data() {
        return {
            content: `<p>welcome to use</p>
            editorOption: {
              toolbar:[]
            }
        }
    },
    methods: {
        onEditorReady(editor) { // 准备编辑器
 
        },
        onEditorBlur(){}, // 失去焦点事件
        onEditorFocus(){}, // 获得焦点事件
        onEditorChange({quill,html,text}){
          // quill:对应实例
          // html:对应html结构
          // text:对应文本内容
        }, // 内容改变事件
    },
    computed: {
        editor() {
            return this.$refs.myQuillEditor.quill;
        },
    },
    created() {
        // ...请求，拿到后台返回的内容对content进行赋值
    }
}
</script>

4.配置你需要的功能

toolbar:[
  ['bold', 'italic', 'underline', 'strike'],    //加粗，斜体，下划线，删除线
  ['blockquote', 'code-block'],     //引用，代码块
  [{ 'header': 1 }, { 'header': 2 }],        // 标题，键值对的形式；1、2表示字体大小
  [{ 'list': 'ordered'}, { 'list': 'bullet' }],     //列表
  [{ 'script': 'sub'}, { 'script': 'super' }],   // 上下标
  [{ 'indent': '-1'}, { 'indent': '+1' }],     // 缩进
  [{ 'direction': 'rtl' }],             // 文本方向
  [{ 'size': ['small', false, 'large', 'huge'] }], // 字体大小
  [{ 'header': [1, 2, 3, 4, 5, 6, false] }],     //几级标题
  [{ 'color': [] }, { 'background': [] }],     // 字体颜色，字体背景颜色
  [{ 'font': [] }],     //字体
  [{ 'align': [] }],    //对齐方式
  ['clean'],    //清除字体样式
  ['image','video']    //上传图片、上传视频
]

一切准备就绪，说下我在富文本编辑器中需要做的功能，在我富文本编辑器上有一排按钮，当点击对应按钮的时候在富文本输入栏光标停止的位置中显示按钮中对应的信息且颜色要高亮

  • 当时我的想法是在失焦事件中拿到对应文本然后拼接内容然后重新赋值data中content的值，发现并不理想，面向百度编程发现居然也有相类似的需求，直接可以拿到光标位置

直接上代码

  // 拿到按钮中的文本信息
  getContent(data) {
    this.getCursorPosition();
    this.setCharacter(data.name, this.curTxtPoint);
  },
  //获取当前模板文本域的光标位置
  getCursorPosition() {
    this.curTxtPoint = this.$refs.myQuillEditor.quill.selection.savedRange.index;
  },
  //插入文本的方式
  setCharacter(str, index) {
    this.curTxtPoint = "";
    this.$refs.myQuillEditor.quill.insertText(index, str, { color: "red" }); // 插入时，可以随意改变插入的文本颜色
  },

5.整体代码如下

<template>
  <div class="content">
    <div class="tags">
      <ul class="operation">
        <li
          class="item"
          v-for="item in operationList"
          :key="item.id"
          @click="getContent(item)"
        >{{item.name}}
        </li>
      </ul>
    </div>
    <div class="rich_editor">
      <quill-editor
      v-model="content" // 编辑框中的内容
      ref="myQuillEditor"
      :options="editorOption" // 编辑器toolbar配置项
      @focus="onEditorFocus($event)" // 聚焦事件
      @blur="onEditorBlur($event)" // 失焦事件
      @change="onEditorChange($event)"> // 改变事件
    </quill-editor>
    </div>
  </div>
</template>
<script>
import { quillEditor } from "vue-quill-editor"; //调用编辑器
import 'quill/dist/quill.core.css';
import 'quill/dist/quill.snow.css';
import 'quill/dist/quill.bubble.css';
export default {
    components: {
        quillEditor
    },
    data() {
        return {
            content: `<p>welcome to use</p>,
            editorOption: {
              toolbar:[
                ['bold', 'italic', 'underline', 'strike'],    //加粗，斜体，下划线，删除线
                ['blockquote', 'code-block'],     //引用，代码块
                [{ 'header': 1 }, { 'header': 2 }],        // 标题，键值对的形式；1、2表示字体大小
                [{ 'list': 'ordered'}, { 'list': 'bullet' }],     //列表
                [{ 'script': 'sub'}, { 'script': 'super' }],   // 上下标
                [{ 'indent': '-1'}, { 'indent': '+1' }],     // 缩进
                [{ 'direction': 'rtl' }],             // 文本方向
                [{ 'size': ['small', false, 'large', 'huge'] }], // 字体大小
                [{ 'header': [1, 2, 3, 4, 5, 6, false] }],     //几级标题
                [{ 'color': [] }, { 'background': [] }],     // 字体颜色，字体背景颜色
                [{ 'font': [] }],     //字体
                [{ 'align': [] }],    //对齐方式
                ['clean'],    //清除字体样式
                ['image','video']    //上传图片、上传视频
              ]
            },
            operationList: [
            { id: 1, name: "#内容一#" },
            { id: 2, name: "#内容二#" },
            { id: 3, name: "#内容三#" },
            { id: 4, name: "#内容四#" },
          ],
        }
    },
    methods: {
      onEditorReady(editor) { // 准备编辑器

      },
      onEditorBlur(){}, // 失去焦点事件
      onEditorFocus(){}, // 获得焦点事件
      onEditorChange({quill,html,text}){
        // quill:对应实例
        // html:对应html结构
        // text:对应文本内容

        this.content = html;
      }, // 内容改变事件
      getContent(data) {
        this.getCursorPosition();
        this.setCharacter(data.name, this.curTxtPoint);
      },
      getCursorPosition() {
          this.curTxtPoint = this.$refs.myQuillEditor.quill.selection.savedRange.index;
      },
      //插入文本的方式
      setCharacter(str, index) {
        this.curTxtPoint = "";
        this.$refs.myQuillEditor.quill.insertText(index, str, { color: "red" });
      },
    },
    computed: {
        editor() {
            return this.$refs.myQuillEditor.quill;
        },
    },
    created() {
        // ...请求，拿到后台返回的内容对content进行赋值
    }
}
</script>
